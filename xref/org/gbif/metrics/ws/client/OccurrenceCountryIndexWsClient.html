27    private static final GenericType<Map<Country, Long>> GENERIC_TYPE = new GenericType<Map<Country, Long>>() {
28    };

30    @Inject
31    public OccurrenceCountryIndexWsClient(WebResource resource) {
32      super(resource);
33    }

35    @Override
36    public Map<Country, Long> publishingCountriesForCountry(Country country) {
37      return getRequest(QueryParamBuilder.create("country", country.getIso2LetterCode()).build(),
38        "occurrence/counts/publishing_countries");
39    }
40  
41    @Override
42    public Map<Country, Long> countriesForPublishingCountry(Country publishingCountry) {
43      return getRequest(QueryParamBuilder.create("publishingCountry", publishingCountry.getIso2LetterCode()).build(),
44        "occurrence/counts/countries");
45    }
46  
47    /**
48     * Executes a get request whose returned value is a SortedMap<Country, Integer>.
49     */
50    private Map<Country, Long> getRequest(MultivaluedMap<String, String> params, String path) {
51      final Map<Country, Long> res = get(GENERIC_TYPE, params, path);
52      return ImmutableSortedMap.copyOf(res,
53        Ordering.natural().onResultOf(Functions.forMap(res)).compound(Ordering.natural()).reverse());
54    }
