21  /**
22   * The callback used from the backfill process to spawn the job to write the new
23   * data in the cube. This is a simple 1 pass scan of the occurrence data which
24   * writes the cube.
25   */
26  class BackfillCallback implements HBaseBackfillCallback {
27  
28    private void addFieldToScan(Scan scan, FieldName fn) {
29      scan.addColumn(Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(fn).getColumnFamilyName()),
30        Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(fn).getColumnName()));
31    }
32  
33    @Override
34    public void backfillInto(Configuration conf, byte[] table, byte[] cf, long snapshotFinishMs) throws IOException {
35      conf = HBaseConfiguration.create(conf);
36  
37      try {
38        Job job = new Job(conf, "Dataset by nub cube writer");
39        job.setJarByClass(TableReaderMapper.class); // required to set up MR classpath
40        // These are set high, as any failure means inaccurate cube counts
41        job.getConfiguration().set("mapred.task.timeout", "1800000"); // 30 mins
42        job.getConfiguration().set("hbase.regionserver.lease.period", "1800000");
43        // We are doing a huge amount of very tiny PUTs so reduce the buffer, as
44        // the default of 2MB means HBase
45        // is being asked to process huge numbers of rows, and tasks start to
46        // timeout. These will likely hit
47        // 1 region if the cube is initially empty.
48        job.getConfiguration().set("hbase.client.write.buffer", "262144"); // 256k
49  
50        TableMapReduceUtil.initTableMapperJob(conf.get(HBaseSourcedBackfill.KEY_SOURCE_TABLE), getScanner(conf),
51          TableReaderMapper.class, Key.class, IntWritable.class, job);
52        job.setCombinerClass(Combiner.class); // to reduce transfer, combine on
53                                              // the map emission
54        job.setReducerClass(CubeWriterReducer.class);
55        job.setNumReduceTasks(conf.getInt(HBaseSourcedBackfill.KEY_NUM_REDUCERS,
56          HBaseSourcedBackfill.DEFAULT_NUM_REDUCERS));
57        job.setOutputKeyClass(NullWritable.class);
58        job.setOutputValueClass(NullWritable.class);
59        job.setOutputFormatClass(NullOutputFormat.class);
60        if (!job.waitForCompletion(true)) {
61          throw new IOException("Unknown error with job.  Check the logs.");
62        }
63  
64      } catch (Exception e) {
65        throw new IOException(e);
66      }
67    }
68  
69    private Scan getScanner(Configuration conf) {
70      Scan scan = new Scan();
71      scan.setCaching(conf.getInt(HBaseSourcedBackfill.KEY_SCANNER_CACHE, HBaseSourcedBackfill.DEFAULT_SCANNER_CACHE));
72      scan.setCacheBlocks(false); // not needed for efficient scanning
73      // Optimize the scan by bringing back only what the TableReaderMapper wants
74      addFieldToScan(scan, FieldName.I_KINGDOM_ID);
75      addFieldToScan(scan, FieldName.I_PHYLUM_ID);
76      addFieldToScan(scan, FieldName.I_CLASS_ID);
77      addFieldToScan(scan, FieldName.I_ORDER_ID);
78      addFieldToScan(scan, FieldName.I_FAMILY_ID);
79      addFieldToScan(scan, FieldName.I_GENUS_ID);
80      addFieldToScan(scan, FieldName.I_SPECIES_ID);
81      addFieldToScan(scan, FieldName.I_NUB_ID);
82      addFieldToScan(scan, FieldName.DATASET_KEY);
83      return scan;
84    }
