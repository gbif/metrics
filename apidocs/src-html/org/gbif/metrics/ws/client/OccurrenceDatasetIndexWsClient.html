004package org.gbif.metrics.ws.client;
005
006import org.gbif.api.service.occurrence.OccurrenceDatasetIndexService;
007import org.gbif.api.vocabulary.Country;
008import org.gbif.ws.client.BaseWsClient;
009import org.gbif.ws.client.QueryParamBuilder;
010
011import java.util.Map;
012import java.util.SortedMap;
013import java.util.UUID;
014import javax.ws.rs.core.MultivaluedMap;
015
016import com.google.common.base.Functions;
017import com.google.common.collect.ImmutableSortedMap;
018import com.google.common.collect.Ordering;
019import com.google.inject.Inject;
020import com.sun.jersey.api.client.GenericType;
021import com.sun.jersey.api.client.WebResource;
022
023
024/**
025 * A web service client to support the accession of occurrence dataset indexes.
026 */
027public class OccurrenceDatasetIndexWsClient extends BaseWsClient implements OccurrenceDatasetIndexService {
028
029  private static final String DATASETS_PATH = "occurrence/counts/datasets";
030
031  private static final String NUBKEY_PARAM = "nubKey";
032  private static final String COUNTRY_PARAM = "country";
033
034  private static final GenericType<Map<UUID, Integer>> GENERIC_TYPE = new GenericType<Map<UUID, Integer>>() {
035  };
036
037  @Inject
038  public OccurrenceDatasetIndexWsClient(WebResource resource) {
039    super(resource);
040  }
041
042  @Override
043  public SortedMap<UUID, Integer> occurrenceDatasetsForCountry(Country country) {
044    return getRequest(QueryParamBuilder.create(COUNTRY_PARAM, country.getIso2LetterCode()).build(),
045      DATASETS_PATH);
046  }
047
048  @Override
049  public SortedMap<UUID, Integer> occurrenceDatasetsForNubKey(int nubKey) {
050    return getRequest(QueryParamBuilder.create(NUBKEY_PARAM, nubKey).build(), DATASETS_PATH);
051  }
052
053  /**
054   * Executes a get request whose returned value is a SortedMap<UUID, Integer>.
055   */
056  private SortedMap<UUID, Integer> getRequest(MultivaluedMap<String, String> params, String path) {
057    final Map<UUID, Integer> res = get(GENERIC_TYPE, params, path);
058    return ImmutableSortedMap.copyOf(res,
059      Ordering.natural().onResultOf(Functions.forMap(res)).compound(Ordering.natural()).reverse());
060  }
061}
