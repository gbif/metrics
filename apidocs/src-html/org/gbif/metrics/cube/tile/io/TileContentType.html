001package org.gbif.metrics.cube.tile.io;
002
003/**
004 * For indicating the source of content for a tile.
005 * Note that we use the terminology PUBLISHING_COUNTRY whereas the Occurrence projects use HOST_COUNTRY.
006 * This is because it really is the publishing country and not the hosting country and it is envisaged that 
007 * the occurrence projects will be renamed.
008 * Be aware that calling hashCode() on an ENUM is not consistent across JVMs!
009 */
010public enum TileContentType {
011  ALL(0), TAXON(1), DATASET(2), PUBLISHER(3), COUNTRY(4), NETWORK(5), PUBLISHING_COUNTRY(6);
012
013  private int id;
014
015  TileContentType(int id) {
016    this.id = id;
017  }
018
019  public static TileContentType INSTANCE(int id) {
020    switch (id) {
021      case 1:
022        return TAXON;
023      case 2:
024        return DATASET;
025      case 3:
026        return PUBLISHER;
027      case 4:
028        return COUNTRY;
029      case 5:
030        return NETWORK;
031      case 6:
032        return PUBLISHING_COUNTRY;
033      default:
034        return ALL;
035    }
036  }
037
038  public int getId() {
039    return id;
040  }
041}
