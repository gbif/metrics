001package org.gbif.metrics.cube.bucketer;
002
003import java.util.UUID;
004
005import com.urbanairship.datacube.CSerializable;
006import com.urbanairship.datacube.bucketers.AbstractIdentityBucketer;
007import com.urbanairship.datacube.serializables.BytesSerializable;
008
009/**
010 * Bucketer that serializes UUIDs as 36 bytes of text.
011 */
012public class UUIDBucketer extends AbstractIdentityBucketer<UUID> {
013  // UUID are 36 long
014  public static final int BYTES=36;
015
016  @Override
017  public CSerializable makeSerializable(UUID coord) {
018    return new BytesSerializable(coord.toString().getBytes());
019  }
020}
