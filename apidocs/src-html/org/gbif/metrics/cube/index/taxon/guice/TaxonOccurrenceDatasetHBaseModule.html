001package org.gbif.metrics.cube.index.taxon.guice;
002
003import org.gbif.metrics.cube.CubeHBaseModule;
004import org.gbif.metrics.cube.CubeIo;
005import org.gbif.metrics.cube.CubeIo.Type;
006import org.gbif.metrics.cube.HBaseCubes;
007import org.gbif.metrics.cube.index.common.UuidIntMap;
008import org.gbif.metrics.cube.index.taxon.OccurrenceDatasetCube;
009
010import java.io.IOException;
011import java.util.Properties;
012
013import com.google.inject.Provides;
014import com.google.inject.Singleton;
015import com.urbanairship.datacube.DataCubeIo;
016import org.apache.hadoop.hbase.HBaseConfiguration;
017
018
019/**
020 * Sets up the CubeIO.
021 */
022public class TaxonOccurrenceDatasetHBaseModule extends CubeHBaseModule {
023
024  private static final String PREFIX = "taxon.occurrence_dataset";
025  private HBaseCubes<UuidIntMap> hbaseCube;
026
027  public TaxonOccurrenceDatasetHBaseModule(Properties props) {
028    super(props, PREFIX);
029  }
030
031  @Override
032  protected void configure() {
033    try {
034      hbaseCube =
035        HBaseCubes.newCombiningBatchAsync(OccurrenceDatasetCube.INSTANCE, UuidIntMap.DESERIALIZER, getCubeTable(),
036          getCf(), HBaseConfiguration.create(), getWriteBatchSize());
037    } catch (IOException e) {
038      this.addError(e);
039    }
040  }
041
042  @Singleton
043  @Provides
044  @CubeIo(Type.TAXON_OCCURRENCE_DATASET)
045  public DataCubeIo<UuidIntMap> getHBaseCube() throws IOException {
046    return hbaseCube.getDataCubeIo();
047  }
048}
