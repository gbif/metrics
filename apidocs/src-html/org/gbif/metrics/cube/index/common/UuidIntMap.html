001package org.gbif.metrics.cube.index.common;
002
003import org.gbif.metrics.cube.util.GenericStringIntMapDatumReader;
004
005import java.io.ByteArrayInputStream;
006import java.io.ByteArrayOutputStream;
007import java.io.IOException;
008import java.util.Map;
009import java.util.Map.Entry;
010import java.util.UUID;
011
012import com.google.common.annotations.VisibleForTesting;
013import com.google.common.base.Objects;
014import com.google.common.base.Preconditions;
015import com.google.common.collect.ImmutableMap;
016import com.google.common.collect.Maps;
017import com.google.common.io.Closeables;
018import com.google.common.io.Flushables;
019import com.urbanairship.datacube.Deserializer;
020import com.urbanairship.datacube.Op;
021import org.apache.avro.Schema;
022import org.apache.avro.Schema.Type;
023import org.apache.avro.generic.GenericDatumWriter;
024import org.apache.avro.io.Decoder;
025import org.apache.avro.io.DecoderFactory;
026import org.apache.avro.io.Encoder;
027import org.apache.avro.io.EncoderFactory;
028import org.apache.avro.util.Utf8;

030
031/**
032 * A simple data structure that can serialize and deserialize itself efficiently using Avro.
033 * This implements the the Op interface to support its use as the target of an address in a datacube,
034 * in addition to a nested deserializer. For safety, this ensures that the integer is always positive
035 * thus any attempts to subtract which produce negative numbers are considered illegal and the map
036 * will remain at 0.
037 */
038public class UuidIntMap implements Op {
039
040  /**
041   * Suitable for use in cube definitions, deserializes using avro.
042   */
043  public static class CountByDatasetDeserializer implements Deserializer<UuidIntMap> {
044
045    @Override
046    public UuidIntMap fromBytes(byte[] bytes) {
047      ByteArrayInputStream bais = new ByteArrayInputStream(bytes);
048      try {
049        GenericStringIntMapDatumReader reader = new GenericStringIntMapDatumReader(schema);
050        Decoder decoder = DecoderFactory.get().binaryDecoder(bais, null);
051        Map<CharSequence, Integer> r = reader.read(null, decoder);
052        return new UuidIntMap(r);
053      } catch (IOException e) {
054        throw new RuntimeException("Unable to deserialize UuidIntMap: " + e.getMessage());
055      } finally {
056        Closeables.closeQuietly(bais);
057      }
058    }
059  }
060
061  public static final CountByDatasetDeserializer DESERIALIZER = new CountByDatasetDeserializer();
062
063  public static final UuidIntMap EMPTY_MAP = new UuidIntMap(ImmutableMap.<CharSequence, Integer>of());
064  private static final Schema schema = Schema.createMap(Schema.create(Type.INT));
065
066  // CharSequence due to Avro format
067  private final Map<CharSequence, Integer> counts;
068
069  /**
070   * Internal constructor for use with the format used by Avro.
071   * Prefer the static factory over this.
072   */
073  @VisibleForTesting
074  UuidIntMap(Map<CharSequence, Integer> counts) {
075    this.counts = ImmutableMap.copyOf(counts); // defensive copy
076  }
077
078  /**
079   * The preferred static factory to use.
080   */
081  public static UuidIntMap newInstance(Map<UUID, Integer> counts) {
082    Map<CharSequence, Integer> c = Maps.newHashMap();
083    for (Entry<UUID, Integer> e : counts.entrySet()) {
084      c.put(e.getKey().toString(), e.getValue());
085    }
086    return new UuidIntMap(c);
087  }
088
089  private static Map<CharSequence, Integer> toCharsequenceIntMap(Map<Utf8, Integer> input) {
090    ImmutableMap.Builder<CharSequence, Integer> builder = new ImmutableMap.Builder<CharSequence, Integer>();
091    for (Entry<Utf8, Integer> entry : input.entrySet()) {
092      builder.put(entry.getKey().toString(), entry.getValue());
093    }
094    return builder.build();
095  }
096
097  /**
098   * Merges the mapped counts in the other op.
099   */
100  @Override
101  public UuidIntMap add(Op otherOp) {
102    Preconditions.checkArgument(otherOp instanceof UuidIntMap, "Can only add type UuidIntMap");
103    UuidIntMap o = (UuidIntMap) otherOp;
104    return new UuidIntMap(merge(counts, o.getCounts(), false));
105  }
106
107  @Override
108  public boolean equals(Object obj) {
109    if (this == obj) {
110      return true;
111    }
112    if (!(obj instanceof UuidIntMap)) {
113      return false;
114    }
115    UuidIntMap other = (UuidIntMap) obj;
116    // we don't care on ordering, so we just need to ensure there are no
117    // entries that differ
118    return counts != null && other.counts != null && Maps.difference(counts, other.counts).entriesDiffering().isEmpty();
119  }
120
121  public Map<CharSequence, Integer> getCounts() {
122    return counts;
123  }
124
125  @Override
126  public int hashCode() {
127    return Objects.hashCode(counts);
128  }
129
130  /**
131   * Serializes using avro.
132   */
133  @Override
134  public byte[] serialize() throws RuntimeException {
135    ByteArrayOutputStream baos = new ByteArrayOutputStream();
136    try {
137      Encoder e = EncoderFactory.get().binaryEncoder(baos, null);
138      GenericDatumWriter<Map<CharSequence, Integer>> writer =
139        new GenericDatumWriter<Map<CharSequence, Integer>>(schema);
140      writer.write(counts, e);
141      Flushables.flushQuietly(e);
142      return baos.toByteArray();
143    } catch (IOException e) {
144      // We can't throw a checked exception, so repackage
145      throw new IllegalStateException("Unable to serialize: " + e);
146    } finally {
147      Closeables.closeQuietly(baos);
148    }
149  }
150
151  /**
152   * Merges the mapped counts in the other op.
153   */
154  @Override
155  public UuidIntMap subtract(Op otherOp) {
156    Preconditions.checkArgument(otherOp != null && otherOp instanceof UuidIntMap, "Can only subtract type UuidIntMap");
157    UuidIntMap o = (UuidIntMap) otherOp;
158    return new UuidIntMap(merge(counts, o.getCounts(), true));
159  }
160
161  @Override
162  public String toString() {
163    return Objects.toStringHelper(this)
164      .add("counts", counts)
165      .toString();
166  }
167
168  /**
169   * Subtraction aware merge.
170   * 
171   * @param m1 The source
172   * @param m2 To be merged into m1
173   * @param isSubtract If true, m2 is subtracted from m1
174   */
175  private Map<CharSequence, Integer> merge(Map<CharSequence, Integer> m1,
176    Map<CharSequence, Integer> m2, boolean isSubtract) {
177    Map<CharSequence, Integer> source = Maps.newHashMap(m1);
178    for (Entry<CharSequence, Integer> e : m2.entrySet()) {
179      CharSequence k = e.getKey();
180      int i = e.getValue();
181      if (isSubtract) {
182        if (source.containsKey(k)) {
183          i = source.get(k) - i;
184        } else {
185          continue; // ignore subtractions if not in the source
186        }
187      } else {
188        i = source.containsKey(k) ? source.get(k) + i : i;
189      }
190      i = i < 0 ? 0 : i; // ensure we don't go below 0
191      source.put(k, i);
192    }
193    return ImmutableMap.copyOf(source);
194  }
195}
