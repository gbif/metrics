003import org.gbif.api.service.occurrence.OccurrenceDatasetIndexService;
004import org.gbif.metrics.ws.client.guice.MetricsWsClientModule;
005import org.gbif.metrics.ws.guice.MetricsWsServletListener;

009
010import java.io.IOException;
011
012import com.google.inject.Guice;
013import com.google.inject.Injector;
014import org.junit.Before;
015import org.junit.Ignore;
016import org.junit.Test;
017
018/**
019 * A simple Grizzly based WS IT test
020 */
021@Ignore
022// TODO: test failing for incompatible javax version?
023public class OccurrenceDatasetIndexWsClientIT extends BaseResourceTest {
024
025  // should this change then we are really in trouble
026  private static final int ANIMALIA_KEY = 1;
027  private static final String CONTEXT = "metrics";
028  private static final String PROPERTIES_FILE = "metrics.properties";
029  private OccurrenceDatasetIndexService wsClient; // under test
030
031  public OccurrenceDatasetIndexWsClientIT() {
032    super("org.gbif.metrics.ws.resources", CONTEXT, MetricsWsServletListener.class);
033  }
034
035  /**
036   * Ensures that the read works without throwing exception.
037   */
038  @Test
039  public void basicLookup() {
040    wsClient.occurrenceDatasetsForNubKey(ANIMALIA_KEY);
041  }
042
043  @Before
044  public void init() throws IOException {
045    // This uses UrlBindingModule to dynamically set the named property (metrics.ws.url) taking into account that the
046    // port might have been set at runtime with system properties (hence the metrics-ws.url is omitted
047    // in the properties)
048    Injector clientInjector = Guice.createInjector(
049      new UrlBindingModule(getBaseURI().toString() + CONTEXT, "metrics.ws.url"),
050      new MetricsWsClientModule(PropertiesUtil.loadProperties(PROPERTIES_FILE)));
051    wsClient = clientInjector.getInstance(OccurrenceDatasetIndexService.class);
052  }
053}
